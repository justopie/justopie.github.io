(defun _FreezeLabelLayer
	 (sStyleType lstStyleData blnFreezeStyle / msg)
    ;;_ Freeze label layer
    ;;_ Copyright Â© 2013-2015 by Richard Lawrence
    (if	(= (type lstStyleData) 'LIST)
      (progn
	(foreach n lstStyleData
	  (if
	    (getvar 'tilemode)
	     (vla-put-freeze (op:layer (nth 0 n)) :vlax-true)
	     (vl-cmdf "_vplayer"
		      "f"
		      (nth 0 n)
		      "Current"
		      ""
	     )
	  )
	  (setq	msg (strcat "\n"
			    sStyleType
			    " Label Style \""
			    (nth 1 n)
			    "\""
			    "\n"
			    sStyleType
			    " Label Layer \""
			    (nth 0 n)
			    "\" was frozen. "
		    )
	  )
	)
      )
    )
    (if	msg
      (princ msg)
    )
  )
